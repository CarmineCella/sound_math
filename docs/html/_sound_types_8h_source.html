30// NB: differences with sparkle:
31// output hop is variable thus corrupting audio quality during stretches
32// there are no phase transformations
33// thres is no traditional xsynth
34// there is no fft padding

36// BRUNO - REMATCH
37//1 - calcoli media Mi e stddev Si di√¨ ogni descrittore i nel tuo database
38//2 - registri un pezzettino del segnale che ti arriva in input, in modo che sia il piu significativo possibile
39//3 - di questo segnale, calcoli media M e stddev S
40//4 - fissi un coefficiente di mapping C (che puoi cambiare in tempo reale) che vari ad esempio tra 0.1 e 2 (tanto per fare un esempio)
41//5 - al momento del match, scali ogni descrittore di input Xi tra (Mi - C*Si) e (Mi + C*Si) in modo che piu o meno il segnale sia scalato in [-1,1]
42//6 - poi fai il match con i descrittori, anch'essi scalati in maniera analoga
43//
44//
45
46// PROB MERGE MATLAB
47// I re-arrange the target and source transition matrices so that the same rows/columns
48// correspond to the closest types/centroids. If there are no matches for a type,
49// a new row/column of zeros is inserted to the opposite matrix. Once both matrix
50// have the same size and have been re-arranged, the weighted sum is taken (M=a*T+(1-a)*S)

789 void makeDictionary (T* data, int totSamp, int nchnls) {
790
791 // types building
792 for (int t = 0; t < (int) p->clusters; ++t) {
793 // compute distances from the center of the cluster
794 std::vector <T> distances (p->classes[t].size ());
795 T totDistance = 0;
796
797 for (int j = 0 ; j < p->classes[t].size (); ++j) {
798 std::vector <T> features;
799 if (p->dimensions == 0) { // pca not applied
800 for (unsigned int i = 0 ; i < p->descriptors.size (); ++i) {
801 if (p->switches[i] == true) {
802 features.push_back (p->matrix[i][p->classes[t][j]]);
803 }
804 }
805 } else { // pca applied
806 for (unsigned int i = 0; i < p->pca.size (); ++i) {
807 int c = (int) p->classes[t][j];

810 }
811 }
812
813 T d = p->distFun (&features[0], p->centroids[t], features.size ());
814 distances[j] = d;
815 totDistance += d;
816 }
817
818 if (p->algorithm == Features::RANDOM) {
819 if (p->classes[t].size ()) { // skip empty centriods
820 int rv = (int) rand ();
821 T qr = (T) rv / RAND_MAX;
822 qr *= p->classes[t].size ();
823 int q = (int) qr;
824 int pointer = (int) ((T) p->hop * p->classes[t][q]);
825 int r = pointer + p->N > (totSamp) ? pointer + p->N - (totSamp) : 0;
826 pointer -= r;
827 for (int i = 0; i < p->N; ++i) {
828 T sample = 0;
829 for (int j = 0; j < nchnls; ++j) {
830 int pos = (nchnls * pointer) + (nchnls * i + j);
831 sample += (T) data[pos];
832 }
833 p->types[t][i] = (sample / nchnls) * window[i];
834 }
835 }
836 } else if (p->algorithm == Features::AVERAGE || p->algorithm == Features::WEIGHTED) {
837 for (int q = 0; q < p->classes[t].size (); ++q) {
838 T cnorm = 1. / p->classes[t].size ();
839
840 T d = fabs (distances[q] / (totDistance > soundmath::EPS ? totDistance : 1));
841 if (p->algorithm == Features::WEIGHTED) cnorm = 1. / (d > soundmath::EPS ? d : 1);
842

845 //int r = p->N > totSamp - pointer ? totSamp - pointer : p->N;
846 int r = pointer + p->N > (totSamp) ? pointer + p->N - (totSamp) : 0;
847 pointer -= r;
848 for (int i = 0; i < p->N; ++i) {
849 T sample = 0;
850 for (int j = 0; j < nchnls; ++j) {
851 sample += (T) data[(nchnls * pointer) + (nchnls * i + j)];
852 }
853 p->types[t][i] += (sample / nchnls * cnorm) * window[i];
854 }
855 }
856 } else if (p->algorithm == Features::CLOSEST) {
857 if (p->classes[t].size ()) { // skip empty centriods
858 int q = 0;
859 soundmath::minimum (&distances[0], distances.size (), q);
860
861 int pointer = (int) ((T) p->hop * p->classes[t][q]);
862 //int r = p->N > totSamp - pointer ? totSamp - pointer : p->N;
863 int r = pointer + p->N > (totSamp) ? pointer + p->N - (totSamp) : 0;
864 pointer -= r;
865 for (int i = 0; i < p->N; ++i) {
866 T sample = 0;
867 for (int j = 0; j < nchnls; ++j) {
868 sample += (T) data[(nchnls * pointer) + (nchnls * i + j)];
869 }
870 p->types[t][i] = (sample / nchnls) * window[i];
