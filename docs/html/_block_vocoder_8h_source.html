90 void process (T* wksp, T* wksp2, T P, T F, T D, T I, int C, T threshold, T cross) {
91 fftshift<T> (wksp, m_NN);
92
93 transform (wksp);
94 rect2pol<T> (wksp, m_N);
95
96 // -------------------------------- CROSS SYNTHESIS/MORPHING ---------------------------
97 switch (m_crossMod) {
98 case 1:
99 fftshift<T> (wksp2, m_NN);
100
101 transform (wksp2);
102 rect2pol<T> (wksp2, m_N);
103 for (int i = 0; i < m_N; ++i) {
104 // method 0: Amp(SO) = Amp(S1) + sqrt (cross* Amp(S1) * Amp(S2)) Pha(SO) = Pha(S2)
105 //wksp[2 * i] = (1. - cross) * wksp[2 * i] + cross * sqrt (wksp[2 * i] * wksp2[2 * i]);
106 //wksp[2 * i] = wksp[2 * i] + sqrt (cross * wksp[2 * i] * wksp2[2 * i]);
107 //wksp[2 * i + 1] = wksp2[2 * i + 1];
108
109 // method 1: Amp(SO) = sqrt (Amp(S1)*Amp(S2)) Pha(SO) = (1 - cross) * Pha(S1)+ cross* Pha(S2)
110 wksp[2 * i] = sqrt (wksp[2 * i] * wksp2[2 * i]);
111 wksp[2 * i + 1] = (1. - cross) * wksp[2 * i + 1] + cross * wksp2[2 * i + 1];
112
113 //method 2: Amp(SO) = Amp(S1)*Amp(S2) Pha(SO) = Pha(S1)
114 //wksp[2 * i] = wksp[2 * i] * wksp2[2 * i];
115 //wksp[2 * i + 1] = wksp[2 * i + 1];
116
117 //method 3: Amp(SO) = Amp(S1) Pha(SO) = Pha(S1) + Pha(S2)
118 //wksp[2 * i] = wksp[2 * i];
119 //wksp[2 * i + 1] = wksp[2 * i + 1] + wksp2[2 * i + 1];
120 }
121 break;
122 case 2:
123 fftshift<T> (wksp2, m_NN);
124
125 transform (wksp2);
126 rect2pol<T> (wksp2, m_N);
127
128 // computes both envelopes
129 for (int i = 0; i < m_N; ++i) {
130 m_tmpBuff1[2 * i] = log ((T) (wksp[2 * i] / m_NN) + EPS);
131 m_tmpBuff1[2 * i + 1] = 0;
132 }
133 cepstralEnvelope (C, m_tmpBuff1, m_tmpBuff2);
134 for (int i = 0; i < m_N; ++i) {
135 m_tmpBuff1[2 * i] = log ((T) (wksp2[2 * i] / m_NN) + EPS);
136 m_tmpBuff1[2 * i + 1] = 0;
137 }
138 cepstralEnvelope (C, m_tmpBuff1, m_tmpBuff3);
139
140 for (int i = 0; i < m_N; ++i) {
141 // method 4: using envelopes
142 wksp[2 * i] = (1. - cross) * wksp[2 * i] + cross * wksp[2 * i] *
143 ::exp (2. * (m_tmpBuff3[2 *i] - m_tmpBuff2[2 * i]));
144 }
145 break;
146 case 3: // morphing
147 fftshift<T> (wksp2, m_NN);
148
149 transform (wksp2);
150 rect2pol<T> (wksp2, m_N);
151 morph (wksp, wksp2, m_tmpBuff1, (1. - cross), (1. - cross), m_N);
152 for (int i = 0; i < m_N; ++i) {
153 wksp[2 * i] = m_tmpBuff1[2 * i];
154 wksp[2 * i + 1] = m_tmpBuff1[2 * i + 1];
155 }
156 break;
157 }
158
159
160 #ifdef NO_PEAKS_ESTIMATION
161 double stretch = I / D;
162 double omega = 0, phase = 0, delta = 0;
163 for (int i = 0; i < m_N; ++i) {
164 omega = m_sicvt * (double) i * D;
165 phase = wksp[2 * i + 1];
166 delta = omega + princarg (phase - m_oldAnPhi[i] - omega);
167 m_oldAnPhi[i] = phase;
168 m_psi[i] = princarg (m_psi[i] + delta * stretch * P);
169 wksp[2 * i + 1] = m_psi[i];
170 }
171 #else
172 // -------------------------------- PHASE SYNCING ---------------------------
173 // phase increment computation
174 T omega = 0, delta = 0;
175 for (int i = 0; i < m_N2; ++i) {
176 // T f = parabolic<T> (wksp, i, m_N);
177 omega = m_sicvt * (double) i * D;
178 delta = princarg<T> (wksp[2 * i + 1] - m_oldAnPhi[i] - omega);
179 m_phaseInc[i] = P * (omega + delta) / D;
180 }
181
182 // peak picking
183 int peaksNum = locmax2 (wksp, m_N2, m_peaks);
184 if (peaksNum == 0) ++peaksNum;
185
186 memset (m_psi, 0, sizeof (T) * m_N);
187 for (int i = 0; i < peaksNum; ++i) {
188 m_psi[m_peaks[i]] = m_oldSynPhi[m_peaks[i]] + (T) I * m_phaseInc[m_peaks[i]];
189 }
190
191 // phase locking
192 int start = 0;
193 for (int i = 0; i < peaksNum - 1; ++i) {
194 int pk = m_peaks[i];
195 int next_pk = m_peaks[i + 1];
196 int end = (int) round ((T)(pk + next_pk) * .5);
197 T angRotation = m_psi[m_peaks[i]] - wksp[2 * m_peaks[i] + 1];
198 for (int j = start; j <= end; ++j) {
199 if (j != pk) {
200 m_psi[j] = angRotation + wksp[2 * j + 1];
201 }
202 }
203 start = end + 1;
204 }
205
206 // backup current phases for next round
207 for (int i = 0; i < m_N2; ++i) {
208 m_oldAnPhi[i] = wksp[2 * i + 1];
209 m_oldSynPhi[i] = m_psi[i];
210 }
211 #endif
212
213 // -------------------------------- PITCH-SHIFT / ENVELOPE PRESERVATION ---------------------------
214 if ((C && P) || (C && F)) {
215 // compute inverse transposition in wksp2
216 resample (wksp, wksp2, (1. / P) * F);
217
218 // compute cepstrum on the correction
219 for (int i = 0; i < m_N; ++i) {
220 // NB: the division with m_NN is only for NUMERICAL PROBLEMS since the result
221 // should be the same with or without the division (log (a / n) - log (b / n) == log (a) - log (b))
222 wksp2[2 * i] = (log ((T) (wksp2[2 * i] / m_NN) + EPS) - log ((T) (wksp[2 * i] / m_NN) + EPS));
223 // wksp2[2 * i] = log (EPS + wksp2[2 * i] / (wksp[2 * i] + EPS));
224 wksp2[2 * i + 1] = 0.;
225 }
226
227 // outputs: wksp has the corrected spectrum, wksp2 has the envelope
228 cepstralEnvelope (C, wksp2, m_tmpBuff1);
229
230 T norm1 = (P >= 1 ? 1 : 1. / pow (P, COMPENSATION_EXPONENT));
231 T norm2 = (F <= 1 ? 1 : 1. / pow (1. / F, COMPENSATION_EXPONENT));
232
233 // apply correction and amplitude compensation (experimental)
234 for (int i = 0; i < m_N; ++i) {
235 wksp[2 * i] *= (T) std::exp ((T) (2. * m_tmpBuff1[2 * i])) * norm1 * norm2;
236 }
237 }
238
239 resample (wksp, wksp2, P, threshold);
240
241 // -------------------------------- PHASE-BASED EFFECTS ---------------------------
242 // NB: robotization only works if olap is very high, whisperization only
243 // if window len is small
244 switch (m_phiMod) {
245 case 1:
246 for (int i = 0; i < m_N; ++i) {
247 wksp2[2 * i + 1] = (T) 0;
248 }
249 break;
250 case 2:
251 for (int i = 0; i < m_N; ++i) {
252 wksp2[2 * i + 1] = (T) (rand () % m_N);
253 }
254 break;
255 }
256
257 // synthesis
258 pol2rect<T> (wksp2, m_N);
259 itransform (wksp2);
260 fftshift<T> (wksp2, m_NN);
261 }

284 void resample (const T* wksp, T* wksp2, T ratio, T threshold = 0) {
285 // frequency domain resampling
286 memset (wksp2, 0, sizeof (T) * m_NN);
287 int k = 1;
288 int pos = (int) ((T) k * ratio);
289 while (pos < m_N2 && k < m_N * ratio) {
290 //T a = wksp[2 * k];
291 wksp2[2 * pos] = wksp[2 * k] <= threshold ? 0 : wksp[2 * k]; // denoise
292 //wksp2[2 * pos] = a * a / (a + threshold); // denoise
293 wksp2[2 * pos + 1] = m_psi[k];
294 ++k;
295 pos = (int) ((T) k * ratio);
296 }
297 for (int i = 0; i < m_N2; i++) {
298 int pos = (2 * (m_N - i)) - 2;
299 wksp2[pos] = wksp2[2 * i];
300 wksp2[pos + 1] = -wksp2[2 * i + 1];
301 }
302 }
303 void cepstralEnvelope (int C, const T* wksp, T* wksp2) {
304 memset (wksp2, 0, sizeof (T) * m_NN);
305 for (int i = 0; i < m_N; ++i) {
306 wksp2[2 * i] = wksp[2 * i];
307 wksp2[2 * i + 1] = wksp[2 * i + 1];
308 }
309 // inverse transform
310
311 itransform (wksp2);
312 for (int i = 0; i < m_N; ++i) {
313 wksp2[2 * i] /= m_N;
314 wksp2[2 * i + 1] = 0;
315 }
316
317 // liftering
318 wksp2[0] *= .5;
319 for (int i = C + 1; i < m_N; ++i) {
320 wksp2[2 * i] = 0;
321 }
322
323
324 // spectral envelope
325 transform (wksp2);
326 }

328 void morph (T* input1, T* input2, T* output,
329 T morpha, T morphfr, int fftsize) {
330
331 T amp1, amp2, fr1, fr2;
332 T div;
333
334 for (int i = 0; i < fftsize; i += 2){
335 amp1 = input1[i];
336 amp2 = input2[i];
337 output[i] = amp1 + (amp2 - amp1) * morpha;
338
339 if (i) {
340 // interpolate frs
341 fr1 = input1[i + 1];
342 fr2 = input2[i + 1];
343 div = fr1 ? fr2 / fr1 : 0;
344 div = div > 0 ? div : -div;
345 output[i + 1] = (T)(fr1 * pow (div, morphfr));
346 } else {
347 // this is the nyquist frequency band
348 amp1 = input1[i + 1];
349 amp2 = input2[i + 1];
350 output[i + 1] = amp1 + (amp2 - amp1) * morpha;
351 }
352 }
