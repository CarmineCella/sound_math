30float bartlett (float in, int pos, int samples, int slope) {
31 if (pos <= slope) {
32 return ((float) pos / slope) * in;
33 } else if (pos >= (samples - slope)) {
34 return (float) (samples - pos) / slope * in;
35 } else return in;
36}

38float frand (float min, float max) {
39 float f = 0;
40 short r = (short) (rand ());
41 f = fabs ((float) r / 32768);
42 f *= (max - min);
43 f += min;
44 return f;
45}

209 }
