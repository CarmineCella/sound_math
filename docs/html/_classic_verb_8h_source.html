118 void earlyRef (float dur, int density, float diffusion) {
119 if (m_earlyRef) {
120 delete m_earlyRef;
121 }
122 if (dur < 0) throw std::runtime_error ("invalid total duration for early reflections");
123
124 int samples = (int) (dur * m_sr);
125 if (samples < density) samples = density;
126 if (samples <= 0) samples = 1;
127
128 T* c = new T[samples];
129 T* d = new T[samples];
130 for (int i = 0; i < samples; ++i) c[i] = 0.;
131 c[samples - 1] = 1. / exp (1.);
132
133 int sign = 1;
134 for (int i = 0; i < density; ++i) {
135 int r = (int) frand (0, samples);
136 c[r] = (float) sign / exp ((T) r / samples);
137 sign -= sign;
138 }
139
140 Allpass <T> alp (m_sr, dur, diffusion);
141 alp.process (c, d, samples);
142 //for (int i = 0; i < samples; ++i) std::cout << c[i] << std::endl;
143 for (int i = 0; i < samples; ++i) {
144 c[i] += d[i];
145 }
146 m_earlyRef = new FIR<T> (c, samples);
147 delete [] c;
148 delete [] d;
149 }

201 virtual T* process (const T* input, T* output) {
202 for (int j = 0; j < m_size; ++j) {
203 m_delay[j] = input[j] * m_rescale;
204 output[j] = input[j] * m_direct;
205 }
206
207 m_predelay->process (m_delay, m_tmp0, m_size);
208 //m_earlyRef->process (m_tmp0, m_tmp1, m_size);

211 m_combs[i]->process (m_tmp0, m_cbuff[i], m_size);
212 }
213
214 memset (m_tmp2, 0, sizeof (T) * m_size);
215 for (unsigned int i = 0; i < m_combs.size (); ++i) {
216 for (int j = 0; j < m_size; ++j) {
217 m_tmp2[j] += m_cbuff[i][j];
218 }
219 }
220
221 m_allpass[0]->process (m_tmp2, m_abuff[0], m_size);
222 for (unsigned int i = 1; i < m_allpass.size (); ++i) {
223 m_allpass[i]->process (m_abuff[i - 1], m_abuff[i], m_size);
224 }
225
226 for (int j = 0; j < m_size; ++j) {
227 output[j] += (m_abuff[m_allpass.size () - 1][j] * m_tail + (m_tmp1[j] * m_early));
228 //output[j] += m_tmp1[j];
229 }
230
231 return output;

263 float frand (float min, float max) {
264 float f = 0;
265 short r = abs (rand ());
266 f = fabs ((float) r / 32768);
267 f *= (max - min);
268 f += min;
269 return f;
