80 void design (Parameters<T>* params, AbstractDesigner<T>* designer) {
81 if (m_oscbank) delete [] m_oscbank;
82 m_oscbank = new OscilDescriptor<T>[params->partials * params->thickness];
83
84 int rpartials = params->partials * params->thickness;
85 T tamp = 0;
86 T fcurr = params->fund;
87 for (int i = 0; i < params->partials && fcurr < m_sr / 2; ++i) {
88 T acurr = 1.;
89 fcurr = designer->design (params->fund, params->coeff, i + 1, acurr);
90
91 for (int k = 0; k < params->thickness; ++k) {
92 m_oscbank[i * params->thickness + k].freq = fcurr + k * (float) rand () / RAND_MAX * fcurr * params->spread;
93 m_oscbank[i * params->thickness + k].amp = acurr;
94
95 tamp += m_oscbank[i * params->thickness + k].amp;
96
97 m_oscbank[i * params->thickness + k].phi = (float) rand () / RAND_MAX;
98 m_oscbank[i * params->thickness + k].panL = (float) rand () / RAND_MAX;
99 m_oscbank[i * params->thickness + k].panR = 1. - m_oscbank[i].panL;
100 }
101 rpartials = (i + 1) * params->thickness;
102 }
103
104 for (int i = 0; i < rpartials; ++i) {
105 m_oscbank[i].amp /= tamp;
106 }
107 m_partials = rpartials;
108
109 m_startSample = (int) (params->start * m_sr);
110 m_samples = (int) (params->dur * m_sr);
111 m_eincr1 = 1. / ((T) m_samples * params->slope);
112 m_eincr2 = 1. / (m_samples - ((T) m_samples * params->slope));
113 m_envLimit = (m_samples * params->slope);
114 m_time = 0;
115 m_env = 0;
116 }

122 void process (T* output, int n) {
123 T sicvt = m_tablen / m_sr;
124 T* l = output;
125 T* r = output + 1;
126 while (n--) {
127 T tmp = 0;
128 OscilDescriptor<T>* osc = m_oscbank;
129 int pcurr = m_partials;
130
131 //m_env = (m_time < m_envLimit ? m_env += m_eincr1 : m_env -= m_eincr2);
132 //m_env = hanning (m_time, m_samples);
133 m_env = blackmann (m_time, m_samples);
134
135 while (pcurr--) {
136 T incr = osc->freq * sicvt;
137 int pos = (int) osc->phi;
138 T frac = osc->phi - pos;
139
140 T v = m_tab[pos] * (1. - frac) + m_tab[pos + 1] * frac;
141 osc->phi += incr;
142 if (osc->phi >= m_tablen) osc->phi -= m_tablen;
143 v *= osc->amp;
144
145 *l += v * osc->panL * m_env;
146 tmp += v * osc->panR * m_env;
147 ++osc;
148 };
149 l += 2;
150
151 *r += m_dline[m_ptr];
152 m_dline[m_ptr++] = tmp;
153 m_ptr %= m_stereoSpread;
154 r += 2;
155
156 ++m_time;
157 if (m_time > m_samples) m_time -= m_samples;
158 }
159 }

161 T hanning (long pos, long samples) {
162 return .5 - .5 * (cos ((2. * M_PI * (T) pos) / samples));

165 T blackmann (long pos, long samples) {
166 return .42 - .5 * cos ((2. * M_PI * (T) pos) / (samples - 1)) + .08 *
167 cos ((4. * M_PI * (T) pos) / (samples - 1));
168 }
