24 void analyse (T sr, T* data, int len, int msecPos, int N) {
25 int startPos = (int) ((T) msecPos / 1000. * sr); // position in samples
26
27 if (startPos < 1 || startPos > (len - N)) return;
28
29 //std::cout << "start sample = " << startPos << std::endl;
30
31 T* cbuf = new T[2 * N];
32 memset (cbuf, 0, sizeof (T) * N * 2);
33
34 T* win = new T[N];
35 makeWindow<T> (win, N, .5, .5, 0); // hanning
36
37 T* amps = new T[N];
38 T* freq = new T[N];
39
40 for (int i = startPos; i < startPos + N; ++i) {
41 if (i >= len) break;
42 cbuf[2 * (i - startPos)] = data[i] * win[i - startPos];
43 cbuf[2 * (i - startPos) + 1] = 0;
44 }
45
46 fft (cbuf, N, -1);
47 ampFreqParabolic (cbuf, amps, freq, N, sr);
48
49 std::vector<int> maxima;
50 locmax2AmpFreq<T> (amps, freq, N / 2, maxima, 20.);
51
52 //std::cout << "peaks = " << maxima.size () << std::endl;
53
54 m_partials = maxima.size ();
55 if (m_partials <= 0) return;
56
57 if (m_peaks) delete [] m_peaks;
58 m_peaks = new Peak<T>[m_partials];
59
60 T sumAmp = 0;
61 for (int i = 0; i < m_partials; ++i) {
62 m_peaks[i].freq = freq[maxima[i]];
63 m_peaks[i].amp = amps[maxima[i]];
64 sumAmp += amps[maxima[i]];
65 }
66
67 // FIXME: la normalizzazione non e' del tutto corretta poiche' assume
68 // che la nuova fondamentale sia il primo picco....
69 m_freqNorm = m_peaks[0].freq; // before reordering
70 if (m_freqNorm == 0) m_freqNorm += 20;
71
72 sortSpectrum (m_peaks, m_partials);
73 //
74 //for (int i = 0; i < m_partials; ++i) {
75 // std::cout << m_peaks[i].freq << " Hz, " << m_peaks[i].amp << std::endl;
76 //}
77 //sumAmp *= 3; // FIXME
78
79 delete [] cbuf;
80 delete [] win;
81 delete [] amps;
82 delete [] freq;
83 }
84 virtual T design (T f0, T dev, int part, T& amp) {
85 if (part >= m_partials || part < 1 || m_partials == 0) return 0;
86
87 amp = m_peaks[part - 1].amp;
88 return (m_peaks[part - 1].freq / m_freqNorm) * f0 * dev;
89 }
90 const char* type () const {
91 return "ModelBasedDesigner";
92 }
